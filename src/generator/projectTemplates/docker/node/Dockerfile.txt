FROM node:22.15-bookworm AS builder

RUN apt-get update && \
    apt-get install -y --no-install-recommends \
    build-essential \
        python3 \
        pkg-config \
        librdkafka-dev \
        ca-certificates && \
    rm -rf /var/lib/apt/lists/* \

WORKDIR /usr/src/app
COPY package*.json ./
RUN npm ci || npm install
